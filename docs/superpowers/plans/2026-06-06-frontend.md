# DriveSort Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Vite + React + TypeScript SPA served by the existing FastAPI backend that guides the user through Drive taxonomy building (Setup wizard) and ongoing file review (Scan + Status).

**Architecture:** Single-page app in `web/` directory. Vite dev server proxies `/api` and `/ws` to FastAPI at port 7432. Production build is served as StaticFiles from FastAPI. All Drive mutations are deferred until `/api/stage/commit`.

**Tech Stack:** React 18, TypeScript, Vite 5, Tailwind CSS v3, Radix UI primitives (for a11y), Plotly.js (raw API via ref), @dnd-kit/core (drag-and-drop), React Router v6, Vitest + Testing Library

import type {
  AuthStatus, AnalysisStatus, AnalysisResult,
  TaxonomyNode, AddNodePayload, PatchNodePayload, ConfirmPayload,
  DraftState, SaveDraftPayload,
  StageEntry,
  ScanQueueItem, CorrectPayload,
  CacheStatus,
} from './types'

async function get<T>(url: string): Promise<T> {
  const r = await fetch(url)
  return r.json() as Promise<T>
}

async function post<T>(url: string, body?: unknown): Promise<T> {
  const r = await fetch(url, {
    method: 'POST',
    ...(body !== undefined
      ? { headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }
      : {}),
  })
  return r.json() as Promise<T>
}

async function put<T>(url: string, body: unknown): Promise<T> {
  const r = await fetch(url, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  })
  return r.json() as Promise<T>
}

async function patch<T>(url: string, body: unknown): Promise<T> {
  const r = await fetch(url, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  })
  return r.json() as Promise<T>
}

async function del<T>(url: string): Promise<T> {
  const r = await fetch(url, { method: 'DELETE' })
  return r.json() as Promise<T>
}

function encodePath(path: string) {
  return encodeURIComponent(path)
}

export const api = {
  auth: {
    status: () => get<AuthStatus>('/api/auth/status'),
    login:  () => get<{ auth_url: string }>('/api/auth/login'),
  },

  analysis: {
    trigger: () => post<{ status: string }>('/api/analysis/trigger'),
    status:  () => get<AnalysisStatus>('/api/analysis/status'),
    result:  () => get<AnalysisResult>('/api/analysis/result'),
  },

  taxonomy: {
    list:    ()              => get<TaxonomyNode[]>('/api/taxonomy/nodes'),
    add:     (p: AddNodePayload) => post<TaxonomyNode>('/api/taxonomy/nodes', p),
    patch:   (path: string, p: PatchNodePayload) => patch<TaxonomyNode>(`/api/taxonomy/nodes/${encodePath(path)}`, p),
    delete:  (path: string) => del<{ deleted: string }>(`/api/taxonomy/nodes/${encodePath(path)}`),
    confirm: (path: string, p: ConfirmPayload) => post<{ ok: boolean }>(`/api/taxonomy/nodes/${encodePath(path)}/confirm`, p),
  },

  draft: {
    get:     () => get<DraftState | null>('/api/draft'),
    save:    (p: SaveDraftPayload) => put<{ saved: boolean }>('/api/draft', p),
    discard: () => del<{ discarded: boolean }>('/api/draft'),
  },

  stage: {
    list:   () => get<StageEntry[]>('/api/stage'),
    commit: () => post<{ committed: number }>('/api/stage/commit'),
  },

  scan: {
    trigger: () => post<{ status: string }>('/api/scan/trigger'),

    accept:  (fileId: string) => post<{ accepted: string; path: string | null }>(`/api/scan/queue/${fileId}/accept`),

- [ ] **Step 6: Create `web/src/lib/useWebSocket.ts`**

```ts
import { useEffect, useRef } from 'react'
import type { WsEvent } from './types'

const WS_URL = `${window.location.protocol === 'https:' ? 'wss' : 'ws'}://${window.location.host}/ws`

export function useWebSocket(onEvent: (e: WsEvent) => void) {
  const wsRef    = useRef<WebSocket | null>(null)
  const handlerRef = useRef(onEvent)
  handlerRef.current = onEvent

  useEffect(() => {
    let cancelled = false

    function connect() {
      if (cancelled) return
      const ws = new WebSocket(WS_URL)
      wsRef.current = ws

      ws.onmessage = (e) => {
        try {
          handlerRef.current(JSON.parse(e.data) as WsEvent)
        } catch { /* malformed frame — ignore */ }
      }

      ws.onclose = () => {
        if (!cancelled) setTimeout(connect, 3000)
      }
    }

    connect()
    return () => {
      cancelled = true
      wsRef.current?.close()
    }
  }, [])
}
```

import { api } from '@/lib/api'

import type { AnalysisPhase, WsEvent, UmapPoint } from '@/lib/types'
import { Button } from '@/components/ui/Button'
import { Progress } from '@/components/ui/Progress'
import { ScatterPlot } from '@/components/ScatterPlot'

interface PhaseInfo {
  label: string
  description: string
}

const PHASES: Record<AnalysisPhase, PhaseInfo> = {
  idle:       { label: 'Waiting',            description: 'Ready to start' },
  fetching:   { label: 'Fetching files',     description: 'Listing files in Google Drive…' },
  embedding:  { label: 'Embedding',          description: 'Computing semantic embeddings…' },
  clustering: { label: 'Clustering',         description: 'Running UMAP + HDBSCAN…' },
  naming:     { label: 'Naming clusters',    description: 'Generating cluster labels with LLM…' },
  complete:   { label: 'Complete',           description: 'Analysis finished' },
  error:      { label: 'Error',              description: 'Something went wrong' },
}

const PHASE_ORDER: AnalysisPhase[] = ['fetching', 'embedding', 'clustering', 'naming', 'complete']

function phaseIndex(p: AnalysisPhase) {
  return PHASE_ORDER.indexOf(p)
}

export function Analyse() {
  const navigate = useNavigate()
  const [phase, setPhase]         = useState<AnalysisPhase>('idle')
  const [progress, setProgress]   = useState(0)
  const [total, setTotal]         = useState(0)
  const [error, setError]         = useState<string | null>(null)
  const [points, setPoints]       = useState<UmapPoint[]>([])
  const [fileCount, setFileCount] = useState<number | null>(null)
  const [cacheHit, setCacheHit]   = useState(false)

  const handleEvent = useCallback((e: WsEvent) => {
    switch (e.type) {
      case 'phase':
        setPhase(e.phase)
        if (e.phase === 'complete') setTimeout(() => navigate('/setup/review'), 1500)
        break
      case 'fetch_complete':
        setTotal(e.total)
        setFileCount(e.total)
        break
      case 'embed_progress':
        setProgress(e.done)
        setTotal(e.total)
        break
      case 'umap_point':
        setPoints(prev => [...prev, { file_id: e.file_id, x: e.x, y: e.y, label: e.label }])
        break
      case 'cluster_cache_hit':
        setCacheHit(true)
        break
      case 'error':
        setError(e.message)
        setPhase('error')
        break
    }
  }, [navigate])

  useWebSocket(handleEvent)

  // Poll status on mount (for reconnect/refresh scenarios)
  useEffect(() => {
    api.analysis.status().then(s => {
      setPhase(s.phase)
      setProgress(s.progress)
      setTotal(s.total)
      if (s.error) setError(s.error)
      if (s.phase === 'complete') navigate('/setup/review', { replace: true })
    }).catch(() => {})
  }, [navigate])

  async function startAnalysis() {
    setPoints([])
    setError(null)
    await api.analysis.trigger()
  }

  const pct = total > 0 ? Math.round((progress / total) * 100) : 0

  return (
    <div className="h-full overflow-y-auto p-6">
      <div className="max-w-3xl mx-auto space-y-6">
        <div>
          <h2 className="text-xl font-semibold text-white">Analyse Files</h2>
          {fileCount !== null && (
            <p className="text-sm text-gray-400 mt-1">{fileCount} files discovered</p>
          )}
          {cacheHit && (
            <p className="text-xs text-cyan mt-1">⚡ Cluster cache hit — skipping UMAP/HDBSCAN</p>
          )}

        <div className="grid gap-3">
          {PHASE_ORDER.filter(p => p !== 'complete').map(p => {

            const curIdx  = phaseIndex(phase)
            const done    = idx < curIdx || phase === 'complete'
            const active  = p === phase
            const info    = PHASES[p]
            return (
              <div
                key={p}
                className={`flex items-center gap-3 rounded-lg border p-4 transition-colors ${
                  active  ? 'border-accent bg-accent/5' :
                  done    ? 'border-border bg-base-800 opacity-60' :

                <div className="flex-1 min-w-0">

## Task 5: ScatterPlot component (Plotly.js)

**Files:**
- Modify: `web/src/components/ScatterPlot.tsx`

ScatterPlot has two modes:
- **Live mode** (`streaming=true`): dots arrive one-by-one via `points` prop updates; uses `Plotly.extendTraces` for efficiency.
- **Static mode** (default): renders a full `points` array at once; re-renders if `points` reference changes; supports click-to-select and lasso.

- [ ] **Step 1: Write the full ScatterPlot implementation**

```tsx
import { useEffect, useRef, useCallback } from 'react'
import type { UmapPoint } from '@/lib/types'

// 20 distinguishable cluster colours; index -1 (outliers) → grey
const PALETTE = [
  '#6366f1', '#22d3ee', '#f59e0b', '#22c55e', '#f43f5e',
  '#8b5cf6', '#14b8a6', '#fb923c', '#a3e635', '#e879f9',
  '#60a5fa', '#34d399', '#fbbf24', '#f87171', '#a78bfa',
  '#2dd4bf', '#fb7185', '#86efac', '#fde68a', '#c4b5fd',
]

function clusterColor(label: number): string {
  if (label < 0) return '#2a2a3a'
  return PALETTE[label % PALETTE.length]
}

interface ScatterPlotProps {
  points: UmapPoint[]
  height?: number
  streaming?: boolean
  onSelect?: (fileIds: string[]) => void
}

// Lightweight Plotly type shim (we import the actual module at runtime)
// eslint-disable-next-line @typescript-eslint/no-explicit-any
type PlotlyMod = any

export function ScatterPlot({ points, height = 400, streaming = false, onSelect }: ScatterPlotProps) {
  const divRef     = useRef<HTMLDivElement>(null)
  const plotlyRef  = useRef<PlotlyMod | null>(null)
  const countRef   = useRef(0)   // tracks how many points have been added in live mode

  const layout = {
    paper_bgcolor: '#080812',
    plot_bgcolor:  '#080812',
    margin:        { l: 0, r: 0, t: 0, b: 0 },
    xaxis:         { showgrid: false, zeroline: false, showticklabels: false, showline: false },
    yaxis:         { showgrid: false, zeroline: false, showticklabels: false, showline: false },
    showlegend:    false,
    dragmode:      onSelect ? 'lasso' : 'zoom',
  }

  const config = {
    responsive: true,
    displayModeBar: false,
    scrollZoom: true,
  }

  // Initial mount: load Plotly and create empty plot
  useEffect(() => {
    let mounted = true
    import('plotly.js-dist-min').then((mod) => {
      if (!mounted || !divRef.current) return
      plotlyRef.current = mod.default ?? mod
      plotlyRef.current.newPlot(
        divRef.current,
        [{ type: 'scattergl', mode: 'markers', x: [], y: [], text: [],
           marker: { size: 6, color: [], opacity: 0.85 } }],
        layout,
        config,
      )
      if (onSelect) {
        divRef.current.on('plotly_selected', (evt: { points: { text: string }[] }) => {
          if (evt?.points) onSelect(evt.points.map(p => p.text))
        })
      }
    })
    return () => { mounted = false }
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [])

  // Streaming mode: extend traces one-by-one as points arrive
  const prevLen = useRef(0)
  useEffect(() => {
    if (!streaming || !plotlyRef.current || !divRef.current) return
    const newPts = points.slice(prevLen.current)
    if (newPts.length === 0) return
    prevLen.current = points.length

    plotlyRef.current.extendTraces(
      divRef.current,
      {
        x:    [newPts.map(p => p.x)],
        y:    [newPts.map(p => p.y)],
        text: [newPts.map(p => p.file_id)],
        'marker.color': [newPts.map(p => clusterColor(p.label))],
      },
      [0],
    )
  }, [points, streaming])

  // Static mode: full redraw when points array changes
  useEffect(() => {
    if (streaming || !plotlyRef.current || !divRef.current) return
    plotlyRef.current.react(
      divRef.current,
      [{ type: 'scattergl', mode: 'markers',
         x:    points.map(p => p.x),
         y:    points.map(p => p.y),
         text: points.map(p => p.file_id),
         marker: { size: 6, color: points.map(p => clusterColor(p.label)), opacity: 0.85 },
      }],
      layout,
      config,
    )
    // Reset counter so streaming can continue if mode changes
    prevLen.current = points.length
    countRef.current = points.length
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [points, streaming])

  return <div ref={divRef} style={{ width: '100%', height }} />
}
```

function slugify(name: string, parent: string | null): string {
  const seg = name.toLowerCase().replace(/\s+/g, '-').replace(/[^a-z0-9-]/g, '')

}

export function TaxonomyTree({
  nodes, selectedPath, draggingFileId,
  onSelectPath, onAddNode, onRenameNode, onDeleteNode, onDropFile,
}: TaxonomyTreeProps) {

  const [editing, setEditing]       = useState<string | null>(null)
  const [editVal, setEditVal]       = useState('')
  const [adding, setAdding]         = useState<string | null>(null)  // parent path | null = root

    setExpanded(prev => {
      const s = new Set(prev)
      s.has(path) ? s.delete(path) : s.add(path)
      return s
    })
  }

  function commitAdd(parent: string | null, name: string) {
    if (!name.trim()) return
    const path = slugify(name.trim(), parent)
    onAddNode(path, name.trim(), parent)
    setAdding(null)
    setAddVal('')
    setRootAdd(false)
    setRootAddVal('')
  }

  function handleDragOver(e: React.DragEvent, nodePath: string) {
    e.preventDefault()
    setDragOver(nodePath)
  }

  function handleDrop(e: React.DragEvent, nodePath: string) {
    e.preventDefault()
    setDragOver(null)
    const fileId = e.dataTransfer.getData('text/plain') || draggingFileId
    if (fileId) onDropFile(fileId, nodePath)
  }

  function renderItems(items: TreeItem[]): React.ReactNode {
    return items.map(item => {
      const path = item.node.path
      const isExpanded = expanded.has(path)
      return (
        <div key={path}>
          <NodeRow
            item={item}
            selected={selectedPath === path}
            isDragOver={dragOver === path}
            isEditing={editing === path}
            editValue={editVal}
            isAdding={adding === path}
            addValue={addVal}
            expanded={isExpanded}
            hasChildren={item.children.length > 0}
            onToggleExpand={() => toggleExpand(path)}
            onSelect={() => onSelectPath(path)}
            onStartRename={() => { setEditing(path); setEditVal(item.node.name) }}
            onEditChange={setEditVal}
            onCommitRename={() => { onRenameNode(path, editVal); setEditing(null) }}
            onCancelRename={() => setEditing(null)}
            onStartAdd={() => { setAdding(path); setAddVal(''); setExpanded(prev => new Set([...prev, path])) }}
            onAddChange={setAddVal}
            onCommitAdd={() => commitAdd(path, addVal)}
            onCancelAdd={() => { setAdding(null); setAddVal('') }}
            onDelete={() => onDeleteNode(path)}
            onDragOver={e => handleDragOver(e, path)}
            onDrop={e => handleDrop(e, path)}
            onDragLeave={() => setDragOver(null)}
          />
          {isExpanded && item.children.length > 0 && renderItems(item.children)}
        </div>
      )
    })
  }

  return (
    <div className="h-full flex flex-col p-2">
      <div className="flex-1 overflow-y-auto">
        {tree.length === 0 && (
          <p className="text-xs text-gray-600 px-3 py-2">No folders yet — add one below.</p>
        )}
        {renderItems(tree)}
      </div>

      {/* Root-level "Add folder" */}
      <div className="border-t border-border pt-2 mt-2">
        {rootAdd ? (
          <div className="flex items-center gap-1 px-2 py-1">
            <Folder className="h-4 w-4 text-accent shrink-0" />
            <input
              autoFocus
              value={rootAddVal}
              onChange={e => setRootAddVal(e.target.value)}
              onKeyDown={e => {
                if (e.key === 'Enter') { commitAdd(null, rootAddVal); setRootAdd(false) }
                if (e.key === 'Escape') { setRootAdd(false); setRootAddVal('') }
              }}
              placeholder="Top-level folder name…"
              className="flex-1 bg-base-700 rounded px-1 text-sm text-white outline-none border border-accent/40"
            />
            <button className="p-0.5 text-green-400" onClick={() => { commitAdd(null, rootAddVal); setRootAdd(false) }}><Check className="h-3 w-3" /></button>
            <button className="p-0.5 text-gray-500" onClick={() => setRootAdd(false)}><X className="h-3 w-3" /></button>
          </div>
        ) : (
          <button
            className="flex items-center gap-1.5 w-full px-3 py-1.5 text-sm text-gray-500 hover:text-white hover:bg-base-700 rounded-md transition-colors"
            onClick={() => setRootAdd(true)}
          >
            <Plus className="h-4 w-4" /> Add folder

export function Scan() {

      .then(([q, n]) => {
        setQueue(q)

"""
drivesort/server.py
-------------------
FastAPI application factory.

Routes:
  /api/auth/*      OAuth flow
  /api/analysis/*  Embed + cluster trigger, WebSocket progress
  /api/taxonomy/*  Taxonomy tree CRUD
  /api/draft/*     Draft load/save/discard
  /api/stage/*     Staged changes + commit
  /api/scan/*      Scan queue, accept, correct
  /api/cache/*     Cache status + invalidation
  /ws              WebSocket
  /                React SPA (static files, production only)
"""
from __future__ import annotations

from pathlib import Path

from fastapi import FastAPI, Request
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import FileResponse
from fastapi.staticfiles import StaticFiles

from .api import auth, analysis, taxonomy, draft, stage, scan, cache, ws as ws_router


def build_app(static_dir: Path | None = None) -> FastAPI:
    application = FastAPI(title="DriveSort", version="0.2.0")

    application.add_middleware(
        CORSMiddleware,
        allow_origins=["http://localhost:5173", "http://localhost:7432"],
        allow_methods=["*"],
        allow_headers=["*"],
    )

    application.include_router(auth.router,      prefix="/api/auth",     tags=["auth"])
    application.include_router(analysis.router,  prefix="/api/analysis", tags=["analysis"])
    application.include_router(taxonomy.router,  prefix="/api/taxonomy", tags=["taxonomy"])
    application.include_router(draft.router,     prefix="/api/draft",    tags=["draft"])
    application.include_router(stage.router,     prefix="/api/stage",    tags=["stage"])
    application.include_router(scan.router,      prefix="/api/scan",     tags=["scan"])
    application.include_router(cache.router,     prefix="/api/cache",    tags=["cache"])
    application.include_router(ws_router.router, tags=["ws"])

    if static_dir is not None and static_dir.is_dir():
        # Serve static assets (JS/CSS bundles)
        application.mount(
            "/assets",
            StaticFiles(directory=static_dir / "assets"),
            name="assets",
        )

        # SPA catch-all: any unmatched GET returns index.html
        index = static_dir / "index.html"

        @application.get("/{full_path:path}", include_in_schema=False)
        async def spa_fallback(request: Request, full_path: str) -> FileResponse:
            return FileResponse(index)

    return application


# Module-level app instance for uvicorn
_static = Path(__file__).parent.parent / "web" / "dist"
